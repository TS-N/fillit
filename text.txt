
void		spitresult(t_tet **head, unsigned int s)
{
	t_tet			*node;
	unsigned int	i;
	unsigned int	scube;
	char			*grid;

	node = *head;
	scube = s * s;
	grid = (char *)malloc(scube);
	gfill(node, grid, scube);
	i = 0;
	while (i < scube)
	{
		if (i % (s - 1) == 0 && i + 1 != scube)
			ft_putchar('\n');
		grid[i] ? ft_putchar(grid[i]) : ft_putchar('.');
		++i;
	}
	if (node)
		ft_putchar('\n');
	free(grid);
	return ;
}
